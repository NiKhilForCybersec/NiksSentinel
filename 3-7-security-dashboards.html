SOC Metrics Dashboard

Key SOC KPIs

# Mean Time to Detect (MTTD)
# Requires tracking when threat started vs when detected
index=notable earliest=-30d
| eval detect_delay = _time - event_time
| stats avg(detect_delay) AS mttd_seconds
| eval mttd_hours = round(mttd_seconds/3600, 2)

# Mean Time to Acknowledge (MTTA)
index=notable status!="New" earliest=-30d
| eval tta = status_time - _time
| stats avg(tta) AS mtta_seconds
| eval mtta_minutes = round(mtta_seconds/60, 2)

# Mean Time to Resolve (MTTR)
index=notable status IN ("Resolved", "Closed") earliest=-30d
| eval ttr = status_time - _time
| stats avg(ttr) AS mttr_seconds
| eval mttr_hours = round(mttr_seconds/3600, 2)

# Mean Time to Contain (MTTC)
# Requires custom field tracking containment time
index=notable containment_time=* earliest=-30d
| eval ttc = containment_time - _time
| stats avg(ttc) AS mttc_seconds
| eval mttc_hours = round(mttc_seconds/3600, 2)

Alert Volume Metrics

# Daily Notable Event Volume
index=notable earliest=-30d
| timechart span=1d count AS notables
| trendline sma7(notables) AS weekly_trend

# Notable Events by Urgency
index=notable earliest=-7d
| stats count by urgency
| eval order = case(urgency=="critical",1, urgency=="high",2, 
                    urgency=="medium",3, urgency=="low",4, 1==1,5)
| sort order
| fields - order

# True/False Positive Rate
index=notable disposition=* earliest=-30d
| stats count(eval(disposition="True Positive")) AS tp,
        count(eval(disposition="False Positive")) AS fp,
        count(eval(disposition="Benign True Positive")) AS btp,
        count AS total
| eval tp_rate = round((tp/total)*100, 1)
| eval fp_rate = round((fp/total)*100, 1)

# Alert Volume by Source
index=notable earliest=-30d
| stats count by rule_name
| sort -count
| head 20

Analyst Performance

# Cases per Analyst
index=notable owner!=unassigned earliest=-30d
| stats count AS cases_handled,
        avg(eval(status_time - _time)) AS avg_handle_time
  by owner
| eval avg_handle_hours = round(avg_handle_time/3600, 2)
| sort -cases_handled

# Resolution Quality
index=notable disposition=* earliest=-30d
| stats count(eval(disposition="True Positive")) AS tp,
        count(eval(disposition="False Positive")) AS fp
  by owner
| eval accuracy = round((tp/(tp+fp))*100, 1)
| sort -accuracy

# Backlog by Analyst
index=notable status="New" OR status="In Progress"
| stats count by owner
| sort -count

Investigation Dashboard

# Entity Investigation Dashboard with Token Inputs
# Set tokens: $entity$ (user or host to investigate)

# Entity Timeline
index=* (user=$entity$ OR host=$entity$ OR src_ip=$entity$ OR dest=$entity$) earliest=-7d
| timechart span=1h count by index

# Authentication Activity
index=security EventCode IN (4624, 4625) user=$entity$ earliest=-7d
| timechart span=1d count(eval(EventCode=4624)) AS success, 
                    count(eval(EventCode=4625)) AS failure

# Process Execution
index=sysmon EventCode=1 (User=$entity$ OR Computer=$entity$) earliest=-7d
| table _time, Computer, User, Image, CommandLine

# Network Connections
index=firewall (src_ip=$entity$ OR dest_ip=$entity$) earliest=-7d
| stats sum(bytes) AS total_bytes, count AS connections by dest_ip, dest_port
| sort -total_bytes

# Risk Events
index=risk risk_object=$entity$ earliest=-30d
| table _time, source, risk_score, risk_message

# Associated Notable Events
index=notable (user=$entity$ OR src=$entity$ OR dest=$entity$) earliest=-30d
| table _time, rule_title, urgency, status_label

    index=notable earliest=-24h

  </query>
  <earliest>-24h</earliest>
  <latest>now</latest>
</search>

<!-- Panel 1: Total count -->
<search base="base_notable_search">
  <query>| stats sum(count) AS total</query>
</search>

<!-- Panel 2: By urgency -->
<search base="base_notable_search">
  <query>| stats sum(count) by urgency</query>
</search>

<!-- Panel 3: By rule -->
<search base="base_notable_search">
  <query>| stats sum(count) by rule_name | sort -sum(count) | head 10</query>
</search>

tstats for Performance

# Use tstats for high-performance dashboards
# Authentication Overview
| tstats summariesonly=true count FROM datamodel=Authentication 
  WHERE * BY Authentication.action, _time span=1h
| timechart span=1h sum(count) by Authentication.action

# Network Traffic Summary
| tstats summariesonly=true sum(All_Traffic.bytes) AS bytes 
  FROM datamodel=Network_Traffic 
  WHERE * BY All_Traffic.dest_port, _time span=1h
| timechart span=1h sum(bytes) by All_Traffic.dest_port limit=10

ğŸ’¼ Interview Question

"What SOC metrics would you track and why?"

Detection Metrics: MTTD (how fast we detect threats)
Response Metrics: MTTA (acknowledgment speed), MTTR (resolution time), MTTC (containment time)
Quality Metrics: True positive rate, false positive rate
Volume Metrics: Alert volume trends, backlog
Coverage Metrics: Data source coverage, detection coverage (MITRE)

These metrics help identify process inefficiencies, staffing needs, detection quality issues, and demonstrate SOC value to leadership.
